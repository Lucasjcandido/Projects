CREATE TABLE TB_PROJETO
{
CD_CODIGO INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(45) NOT NULL,
DATA_INICIO DATE,
HORAS_PREVISTAS VARCHAR(45)
};

CREATE TABLE TB_PROJETO_has_FUNCIONARIO
{
DATA_ALOCACAO DATE,
ID_CODIGO INT,
ID_MATRICULA,
PRIMARY KEY (ID_CODIGO) REFERENCES TB_PROJETO (CD_CODIGO),
PRIMARY KEY (ID_MATRICULA) REFERENCES TB_FUNCIONARIO (CD_MATRICULA)
};

CREATE TABLE TB_FUNCIONARIO
{
CD_MATRICULA PRIMARY KEY AUTO_INCREMENT
NOME VARCHAR(45) NOT NULL,
SEXO VARCHAR(45) NOT NULL,
TELEFONE VARCHAR(45) NOT NULL,
ID_CARGO INT,
ID_DEPARTAMENTO INT,
ID_MATRICULA_CHEFE INT,
PRIMARY KEY (ID_CARGO) REFERENCES TB_CARGO (CD_CODIGO),
PRIMARY KEY (ID_DEPARTAMENTO) REFERENCES TB_DEPARTAMENTO (CD_CODIGO),
PRIMARY KEY (ID_MATRICULA_CHEFE) REFERENCES
? (?)
};

CREATE TABLE DEPARTAMENTO
{
CODIGO INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(45) NOT NULL,
SIGLA VARCHAR(45) NOT NULL
};

CREATE TABLE HISTORICO_FUNCIONAL
{
CODIGO INT PRIMARY KEY AUTO_INCREMENT,
DATE DATE NOT NULL,
HISTORICO LONGTEXT NOT NULL,
ID MATRICULA,
ID CARGO,
PRIMARY KEY (ID_MATRICULA) REFERENCES TB_FUNCIONARIO (CD_MATRICULA),
PRIMARY KEY (ID_CARGO) REFERENCES TB_FUNCIONARIO (ID_CARGO)
};

CREATE TABLE CARGO
{
CODIGO INT PRIMARY KEY AUTO_INCREMENT,
CARGO VARCHAR(45) NOT NULL
};

CREATE TABLE DEPENDENTE
{
CODIGO INT PRIMARY KEY AUTO_INCREMENT,
NOME VARCHAR(45) NOT NULL,
PARENTESCO VARCHAR(45)
PRIMARY KEY (ID_MATRICULA) REFERENCES TB_FUNCIONARIO (CD_MATRICULA)
};
